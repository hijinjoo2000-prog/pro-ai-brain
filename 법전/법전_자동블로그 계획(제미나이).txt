def analyze_top_competitors_for_seo(self):
    self.log(f"🔍 타깃 키워드 '{self.target_keyword}' 네이버 통합검색 상위 3개 블로그 정밀 분석 중...")
    opts = Options()
    opts.add_argument("--headless")
    opts.add_argument("--disable-blink-features=AutomationControlled")
    driver = webdriver.Chrome(service=Service(ChromeDriverManager().install()), options=opts)
    
    target_url = f"https://search.naver.com/search.naver?where=view&query={self.target_keyword}"
    driver.get(target_url)
    time.sleep(3)
    
    soup = BeautifulSoup(driver.page_source, 'html.parser')
    # VIEW 탭의 최상위 블로그 포스트 링크 3개 추출
    posts = soup.select('.api_txt_lines.total_tit')[:3]
    
    if not posts:
        self.log("⚠️ 상위 노출 문서를 찾지 못했습니다. 기본값으로 폴백(Fallback) 모드를 가동합니다.")
        driver.quit()
        return 1500, ["재개발", "입주권", "프리미엄", "투자금", "수익률"]

    total_text = ""
    avg_length = 0
    
    for post in posts:
        link = post.get('href')
        driver.get(link)
        time.sleep(2)
        try:
            # 네이버 블로그는 iframe 내부에 본문이 존재함
            driver.switch_to.frame("mainFrame")
            content = driver.find_element(By.CSS_SELECTOR, ".se-main-container").text
            total_text += content + " "
            avg_length += len(content)
        except Exception as e:
            self.log(f"⚠️ 개별 문서 파싱 오류 통과: {e}")
            
    driver.quit()
    
    avg_length = avg_length // len(posts)
    
    # KoNLPy 형태소 분석기를 이용한 명사 빈도수 추출
    self.log("🧠 수집된 텍스트 빅데이터 형태소 분석 및 LSI 키워드 매핑 진행 중...")
    nouns = self.okt.nouns(total_text)
    
    # 불용어(Stopwords) 처리 및 1글자 단어 제거
    stop_words = ["노량진", "구역", "매물", "부동산", "진짜", "오늘", "저희", "연락", "이", "것", "수"]
    filtered_nouns = [word for word in nouns if len(word) > 1 and word not in stop_words]
    
    count = Counter(filtered_nouns)
    top_15_keywords = [word for word, _ in count.most_common(15)]
    
    self.log(f"📊 경쟁사 문서 평균 글자 수: {avg_length}자")
    self.log(f"📊 SEO 필수 포함 LSI 키워드 그룹: {', '.join(top_15_keywords)}")
    
    return avg_length, top_15_keywords

# [모듈 3] AI 원고 생성 및 자기 주도적 퀄리티 사전 검증 (Scoring)
# =========================================================
def generate_and_validate_ai_content(self, avg_length, top_keywords):
    genai.configure(api_key=GEMINI_API_KEY)
    # 최신 추론 모델 사용
    model = genai.GenerativeModel('gemini-1.5-flash')
    
    for attempt in range(1, 4):
        self.log(f"✍️ AI 원고 작성 및 스코어링 엔진 가동 중... (시도 횟수: {attempt}/3)")
        prompt = f"""
        당신은 대한민국 최고의 재개발 전문 공인중개사입니다. 메인 키워드 '{self.target_keyword}'를 사용하여 심층 분석 블로그 글을 작성하십시오.
        [작성 지침]
        1. 목표 분량: {avg_length}자 내외의 장문.
        2. 반드시 본문 문맥 속에 다음 단어들을 최대한 자연스럽게 모두 포함시킬 것: {', '.join(top_keywords)}
        3. 매물 핵심 데이터: 초기투자금 {self.info['inv']}억, 프리미엄(P) {self.info['premium']}억.
        4. 문체: 단호하고 전문적인 보고서 형식, 그러나 가독성을 높이기 위해 문단을 명확히 나눌 것.
        5. 메인 키워드 '{self.target_keyword}'는 글 전체에 걸쳐 정확히 4회에서 6회 사이만 사용할 것.
        결과물은 오직 본문 텍스트만 출력하세요.
        """
        
        response = model.generate_content(prompt)
        draft = response.text
        
        # [자체 검증 알고리즘 가동]
        missing_keywords = [kw for kw in top_keywords if kw not in draft]
        keyword_density = draft.count(self.target_keyword)
        content_length = len(draft.replace(" ", ""))
        
        validation_passed = True
        
        if len(missing_keywords) > (len(top_keywords) * 0.4): # 60% 이상 포함해야 함
            self.log(f"⚠️ [검증 반려] 필수 LSI 키워드 누락 과다: {', '.join(missing_keywords[:5])} 등")
            validation_passed = False
        
        if keyword_density > 8:
            self.log(f"⚠️ [검증 반려] 메인 키워드 과다 사용으로 인한 스팸 분류 위험 (밀도 초과).")
            validation_passed = False
            
        if content_length < (avg_length * 0.75):
            self.log(f"⚠️ [검증 반려] 문서 볼륨 부족 (현재: {content_length}자 / 목표: {avg_length}자).")
            validation_passed = False
            
        if validation_passed:
            self.log("✅ 원고 검증 완벽 통과. 해당 문서는 상위 3위 이내 노출 알고리즘 기준을 충족합니다.")
            return draft
            
        time.sleep(2) # 재시도 전 API 쿨다운
        
    self.log("❌ 3회 재작성 시도에도 완벽한 요건 충족에 실패하였습니다. 확보된 최선의 초안으로 프로세스를 강제 속행합니다.")
    return draft

def human_typing_emulator(self, driver, element, text):
    action = ActionChains(driver)
    action.move_to_element(element).click().perform()
    time.sleep(0.5)
    for para in text.split('\n'):
        if para.strip() == "":
            action.send_keys(Keys.ENTER).perform()
        else:
            pyperclip.copy(para)

    self.progress_signal.emit(100)
    
    self.finish_signal.emit()
